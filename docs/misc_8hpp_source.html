57 int remaining = 0;
58
59 if (!skip || rhandle.exists("discards")) {
60 try {
61 std::vector<size_t> dims{ static_cast<size_t>(num_cells) };
62 auto dihandle = utils::check_and_open_dataset(rhandle, "discards", H5T_INTEGER, dims);
63
64 if (!skip) {
65 std::vector<int> discards(num_cells);
66 dihandle.read(discards.data(), H5::PredType::NATIVE_INT);
67 for (auto d : discards) {
68 remaining += (d == 0);
69 }
70 } else {
71 remaining = num_cells;
72 }
73
74 } catch (std::exception& e) {
75 throw utils::combine_errors(e, "failed to retrieve discard information from 'results'");
76 }
77 } else {
78 remaining = num_cells;
79 }
80
81 return remaining;
