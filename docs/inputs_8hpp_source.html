66inline ParamDump validate_parameters(const H5::Group& handle, bool embedded, int version) {
67 auto phandle = utils::check_and_open_group(handle, "parameters");
68 ParamDump output;
69
70 // Formats can either be a scalar... or not.
71 std::vector<std::string> formats;
72 output.multi_matrix = false;
73 {
74 auto fhandle = utils::check_and_open_dataset(phandle, "format", H5T_STRING);
75 auto fspace = fhandle.getSpace();
76 if (fspace.getSimpleExtentNdims() == 0) {
77 formats.push_back(utils::load_string(fhandle));
78 } else {
79 if (version < 1001000) {
80 throw std::runtime_error("'format' should be a scalar string in version 1.0");
81 }
82 output.multi_matrix = true;
83 formats = utils::load_string_vector(fhandle);
84 }
85 }
86 output.num_matrices = formats.size();
87
88 auto fihandle = utils::check_and_open_group(phandle, "files");
89 auto nfiles = fihandle.getNumObjs();
90
91 // Checking the runs.
92 std::vector<int> runs;
93 if (output.multi_matrix) {
94 runs = utils::load_integer_vector(phandle, "sample_groups");
95 if (runs.size() != formats.size()) {
96 throw std::runtime_error("'sample_groups' and 'format' should have the same length");
97 }
98
99 int total_files = std::accumulate(runs.begin(), runs.end(), 0);
100 if (total_files != static_cast<int>(nfiles)) {
101 throw std::runtime_error("sum of 'sample_groups' is not equal to the length of 'files'");
102 }
103
104 // Checking that everyone has unique and sorted names.
105 auto names = utils::load_string_vector(phandle, "sample_names");
106 if (names.size() != formats.size()) {
107 throw std::runtime_error("'sample_names' and 'format' should have the same length");
108 }
109
110 if (version >= 2001000) {
111 if (!utils::is_unique_and_sorted(names)) {
112 throw std::runtime_error("duplicated or unsorted values in 'sample_names'");
113 }
114 } else {
115 std::sort(names.begin(), names.end());
116 if (!utils::is_unique_and_sorted(names)) {
117 throw std::runtime_error("duplicated values in 'sample_names'");
118 }
119 }
120 } else {
121 runs.push_back(nfiles);
122 }
123
124 // Checking the files.
125 int sofar = 0;
126 std::vector<std::pair<hsize_t, hsize_t> > bytes;
127 for (size_t r = 0; r < runs.size(); ++r) {
128 auto curf = formats[r];
129 std::vector<std::string> types;
130
131 for (int s = 0; s < runs[r]; ++s, ++sofar) {
132 std::string current = std::to_string(sofar);
133 try {
134 auto curfihandle = utils::check_and_open_group(fihandle, current);
135
136 utils::check_and_open_dataset(curfihandle, "name", H5T_STRING, {});
137 types.push_back(utils::load_string(curfihandle, "type"));
138
139 if (embedded) {
140 bytes.emplace_back(
141 utils::load_integer_scalar<hsize_t>(curfihandle, "offset"),
142 utils::load_integer_scalar<hsize_t>(curfihandle, "size")
143 );
144 } else {
145 utils::check_and_open_dataset(curfihandle, "id", H5T_STRING, {});
146 }
147 } catch (std::exception& e) {
148 throw utils::combine_errors(e, "failed to retrieve information for file " + current);
149 }
150 }
151
152 if (curf == "MatrixMarket") {
153 std::unordered_map<std::string, int> expected;
154 expected["mtx"] = 0;
155 expected["genes"] = 0;
156 expected["annotations"] = 0;
157
158 for (auto t : types) {
159 auto it = expected.find(t);
160 if (it == expected.end()) {
161 throw std::runtime_error("unknown file type '" + t + "' when format is 'MatrixMarket'");
162 }
163 ++(it->second);
164 }
165
166 if (expected["mtx"] != 1) {
167 throw std::runtime_error("expected exactly one 'mtx' file when format is 'MatrixMarket'");
168 }
169 if (expected["genes"] > 1) {
170 throw std::runtime_error("expected no more than one 'genes' file when format is 'MatrixMarket'");
171 }
172 if (expected["annotations"] > 1) {
173 throw std::runtime_error("expected no more than one 'annotation' file when format is 'MatrixMarket'");
174 }
175
176 } else if (curf == "10X") {
177 if (types.size() != 1 || types.front() != "h5") {
178 throw std::runtime_error("expected exactly one 'h5' file when format is '10X'");
179 }
180 } else if (curf == "H5AD") {
181 if (types.size() != 1 || types.front() != "h5") {
182 throw std::runtime_error("expected exactly one 'h5' file when format is 'H5AD'");
183 }
184 }
185 }
186
187 // Checking the files make sense.
188 if (embedded) {
189 hsize_t sofar = 0;
190 for (const auto& b : bytes) {
191 if (b.first != sofar) {
192 throw std::runtime_error("offsets and sizes of 'files' are not sorted and contiguous");
193 }
194 sofar += b.second;
195 }
196 }
197
198 // Checking if there's a batch variable.
199 if (!output.multi_matrix && phandle.exists("sample_factor")) {
200 utils::check_and_open_dataset(phandle, "sample_factor", H5T_STRING, {});
201 output.multi_sample = true;
202 } else {
203 output.multi_sample = output.multi_matrix;
204 }
205
206 // Checking if there's a subset group.
207 output.subset_num = -1;
208
209 if (phandle.exists("subset")) {
210 auto subhandle = utils::check_and_open_group(phandle, "subset");
211
212 if (subhandle.exists("cells")) {
213 auto subcellhandle = utils::check_and_open_group(subhandle, "cells");
214
215 if (subcellhandle.exists("indices")) {
216 auto subidx = utils::load_integer_vector(subcellhandle, "indices");
217
218 for (auto i : subidx) {
219 if (i < 0) {
220 throw std::runtime_error("indices in 'subset/indices' should be non-negative");
221 }
222 }
223
224 if (!utils::is_unique_and_sorted(subidx)) {
225 throw std::runtime_error("indices in 'subset/indices' should be unique and sorted");
226 }
227
228 output.subset_num = subidx.size();
229 } else {
230 utils::check_and_open_dataset(subcellhandle, "field", H5T_STRING, {});
231
232 if (subcellhandle.exists("values")) {
233 auto vhandle = utils::check_and_open_dataset(subcellhandle, "values", H5T_STRING);
234 auto vdims = utils::load_dataset_dimensions(vhandle);
235 if (vdims.size() != 1) {
236 throw std::runtime_error("'subset/values' should be a 1-dimensional string dataset");
237 }
238
239 } else {
240 auto rahandle = utils::check_and_open_dataset(subcellhandle, "ranges", H5T_FLOAT);
241
242 auto radims = utils::load_dataset_dimensions(rahandle);
243 if (radims.size() != 2) {
244 throw std::runtime_error("'subset/ranges' should be a 2-dimensional float dataset");
245 }
246 if (radims[1] != 2) {
247 throw std::runtime_error("'subset/ranges' should have two columns");
248 }
249
250 std::vector<double> loaded(radims[0] * radims[1]);
251 rahandle.read(loaded.data(), H5::PredType::NATIVE_DOUBLE);
252
253 // Should be sorted. Each row is a [start, end) pair,
254 // defining an interval to retain. Intervals should be
255 // non-overlapping and sorted, and HDF5 stores its matrices
256 // as row-order, so start1 <= end1 <= start2 <= end2 <= ...
257 if (!std::is_sorted(loaded.begin(), loaded.end())) {
258 throw std::runtime_error("'subset/ranges' should specify sorted, non-overlapping intervals");
259 }
260 }
261 }
262 }
263 }
264
265 return output;

268inline Details validate_results(const H5::Group& handle, const ParamDump& param_info, int version) {
269 auto rhandle = utils::check_and_open_group(handle, "results");
270 Details output;
271
272 // Pulling out the dimensions and modalities.
273 if (version < 2000000) {
274 output.modalities.push_back("RNA");
275
276 auto dims = utils::load_integer_vector<int>(rhandle, "dimensions");
277 if (dims.size() != 2) {
278 throw std::runtime_error("'dimensions' should be a dataset of length 2");
279 }
280 if (dims[0] < 0 || dims[1] < 0) {
281 throw std::runtime_error("'dimensions' should contain non-negative integers");
282 }
283 output.num_features.push_back(dims[0]);
284 output.num_cells = dims[1];
285 } else {
286 output.num_cells = utils::load_integer_scalar<>(rhandle, "num_cells");
287
288 auto fhandle = utils::check_and_open_group(rhandle, "num_features");
289 size_t nmodals = fhandle.getNumObjs();
290 if (nmodals == 0) {
291 throw std::runtime_error("number of modalities should be positive");
292 }
293
294 for (hsize_t idx = 0; idx < nmodals; ++idx) {
295 std::string modality = fhandle.getObjnameByIdx(idx);
296 output.modalities.push_back(modality);
297 output.num_features.push_back(utils::load_integer_scalar<>(fhandle, modality));
298 }
299 }
300
301 // Checking the number of samples.
302 output.num_samples = 1;
303 if (rhandle.exists("num_samples")) {
304 output.num_samples = utils::load_integer_scalar<int>(rhandle, "num_samples");
305 }
306 if (param_info.multi_matrix) {
307 if (output.num_samples != param_info.num_matrices) {
308 throw std::runtime_error("'num_samples' should be equal to the number of matrices");
309 }
310 } else {
311 if (!param_info.multi_sample && output.num_samples != 1) {
312 throw std::runtime_error("'num_samples' should be 1 for single matrix inputs without 'sample_factor'");
313 }
314 }
315
316 auto check_unique = [](const std::vector<int>& idx, const std::string& msg) -> void {
317 std::unordered_set<int> used;
318 used.reserve(idx.size());
319 for (auto i : idx) {
320 if (i < 0) {
321 throw std::runtime_error(msg + " contains negative values");
322 } else if (used.find(i) != used.end()) {
323 throw std::runtime_error(msg + " contains duplicate values");
324 }
325 used.insert(i);
326 }
327 };
328
329 if (version >= 2000000) {
330 auto ihandle = utils::check_and_open_group<>(rhandle, "identities");
331 for (size_t m = 0; m < output.modalities.size(); ++m) {
332 auto idx = utils::load_integer_vector<int>(ihandle, output.modalities[m]);
333 if (idx.size() != static_cast<size_t>(output.num_features[m])) {
334 throw std::runtime_error("'identities' for modality '" + output.modalities[m] + "' should have length equal to its number of features");
335 }
336 check_unique(idx, "'identities' for modality '" + output.modalities[m] + "'");
337 }
338
339 } else if (version >= 1002000) {
340 auto idx = utils::load_integer_vector<int>(rhandle, "identities");
341 if (idx.size() != static_cast<size_t>(output.num_features[0])) {
342 throw std::runtime_error("'identities' should have length equal to the number of genes");
343 }
344 check_unique(idx, "'identities'");
345
346 } else {
347 if (param_info.multi_matrix) {
348 auto idx = utils::load_integer_vector<int>(rhandle, "indices");
349 if (idx.size() != static_cast<size_t>(output.num_features[0])) {
350 throw std::runtime_error("'indices' should have length equal to the number of genes");
351 }
352 check_unique(idx, "'indices'");
353
354 } else {
355 auto perms = utils::load_integer_vector<int>(rhandle, "permutation");
356 if (perms.size() != static_cast<size_t>(output.num_features[0])) {
357 throw std::runtime_error("'permutation' should have length equal to the number of genes");
358 }
359
360 // Note that the code below implies that all consecutive entries are present,
361 // otherwise we would see duplicates.
362 std::vector<unsigned char> used(perms.size());
363 for (auto p : perms) {
364 if (p < 0 || static_cast<size_t>(p) >= perms.size()) {
365 throw std::runtime_error("'permutation' contains out-of-range values");
366 } else if (used[p]) {
367 throw std::runtime_error("duplicated index in 'permutation'");
368 }
369 used[p] = 1;
370 }
371 }
372 }
373
374 if (param_info.subset_num > -1) {
375 if (param_info.subset_num != output.num_cells) {
376 throw std::runtime_error("inconsistent number of cells in 'parameters/subset/indices' and 'results/num_cells'");
377 }
378 }
379
380 return output;
381}
396inline Details validate(const H5::Group& handle, bool embedded, int version) {
